        \draw[color=purewhite, opacity=\opacity, line width=0.9pt]

            (154.29+\rotation:\radius) -- 
            (205.71+\rotation:\radius) -- 
            (257.14+\rotation:\radius) -- 
            (308.57+\rotation:\radius) -- cycle;
            
        % Overlay Neon Intersections
